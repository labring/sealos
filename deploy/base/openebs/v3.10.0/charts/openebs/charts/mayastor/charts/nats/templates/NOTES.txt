  kubectl exec -n {{ template "nats.namespace" . }} -it deployment/{{ template "nats.fullname" . }}-box -- /bin/sh -l

  nats-box:~# nats sub test &
  nats-box:~# nats pub test hi
  nats-box:~# nc {{ template "nats.fullname" . }} {{ .Values.nats.client.port }}
